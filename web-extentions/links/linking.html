Page-Specific Asset Manager

One file to manage all assets for all your pages.

How This Works

You still only add one file: <script src="asset-manager.js"></script>.

But now, in your page's HTML, you add a *second* small script tag to call the *specific* assets for that page.

On your homepage (e.g., index.html):

<script>loadPageAssets('home');</script>

On your about page (e.g., about.html):

<script>loadPageAssets('about');</script>

Live Demo (Homepage)

We called loadPageAssets('home') on this page.

It loaded Animate.css from the 'common' package and lodash.js from the 'home' package.

The Code for asset-manager.js

Copy this code into a new file named asset-manager.js and link it to all your HTML pages. 

// --- This is the content for asset-manager.js ---

// 1. DEFINE ALL YOUR ASSETS HERE
const pageAssets = {
  // 'common' assets load on EVERY page
  common: {
    css: [
      'https://cdnjs.cloudflare.com/ajax/libs/animate.css/4.1.1/animate.min.css' // Demo: Animate.css
      // 'styles/main-sitewide.css'
    ],
    js: [
      // 'scripts/main-sitewide.js'
    ]
  },
  
  // 'home' assets only load when you call loadPageAssets('home')
  home: {
    css: [
      // 'styles/home-slider.css'
    ],
    js: [
      'https://cdnjs.cloudflare.com/ajax/libs/lodash.js/4.17.21/lodash.min.js' // Demo: lodash
      // 'scripts/home-slider.js'
    ]
  },

  // 'about' assets for the about page
  about: {
    css: [
      // 'styles/timeline.css'
    ],
    js: []
  },

  // 'contact' assets for the contact page
  contact: {
    css: [],
    js: [
      // 'scripts/contact-form-validation.js'
    ]
  }
};

// 2. HELPER FUNCTIONS TO LOAD FILES

/**
 * Creates and appends a CSS <link> tag to the <head>.
 * @param {string} href - The URL of the CSS file.
 */
function loadCss(href) {
  const link = document.createElement('link');
  link.rel = 'stylesheet';
  link.href = href;
  link.onerror = () => console.error(`Failed to load CSS: ${href}`);
  document.head.appendChild(link);
  console.log(`Loading CSS: ${href}`);
}

/**
 * Creates and appends a JS <script> tag to the <body>.
 * @param {string} src - The URL of the JS file.
 */
function loadJs(src) {
  const script = document.createElement('script');
  script.src = src;
  script.defer = true;
  script.onload = () => {
      console.log(`Loaded JS: ${src}`);
      // --- This part is just for the demo to work ---
      if (typeof runDemo === 'function' && src.includes('lodash')) {
          runDemo();
      }
      // --- End of demo-only code ---
  };
  script.onerror = () => console.error(`Failed to load JS: ${src}`);
  document.body.appendChild(script);
  console.log(`Loading JS: ${src}`);
}

// 3. THE MAIN FUNCTION TO CALL FROM YOUR HTML

/**
 * Loads all 'common' assets and all assets for a specific page.
 * @param {string} pageName - The key from pageAssets (e.g., 'home', 'about').
 */
function loadPageAssets(pageName) {
  console.log(`Loading assets for: ${pageName}`);
  
  // Always load common assets
  const common = pageAssets.common;
  if (common) {
    common.css.forEach(loadCss);
    common.js.forEach(loadJs);
  }

  // Load page-specific assets
  const page = pageAssets[pageName];
  if (page) {
    page.css.forEach(loadCss);
    page.js.forEach(loadJs);
  } else {
    console.warn(`No assets found for page: ${pageName}`);
  }
}
