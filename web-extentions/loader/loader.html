Usage

Method 1: Auto-Loading (One Line)

<!-- Default (Loader 3) --> <script src="path/to/loader.js"></script> <!-- Choose specific loader type (1, 2, 3, or 4) --> <script src="path/to/loader.js" data-loader-type="4"></script> <!-- With all options --> <script src="path/to/loader.js" data-loader-type="3" data-loader-size="lg" data-loader-theme="gold" data-loader-min-time="1000"></script>

Method 2: Manual HTML

<!-- Include CSS in head --> <link rel="stylesheet" href="path/to/loader.css"> <!-- Loader 1 --> <div class="loader loader1"> <div><div><div><div><div><div></div></div></div></div></div></div> </div> <!-- Loader 4 (needs 10 nested divs) --> <div class="loader loader4"> <div><div><div><div><div><div><div><div><div><div> </div></div></div></div></div></div></div></div></div></div> </div> <!-- With size and theme --> <div class="loader loader3 loader-lg loader-gold"> <div><div><div><div><div><div></div></div></div></div></div></div> </div>

JavaScript API

// Show the loader (default type 3) VrindaLoader.show(); // Show specific loader type VrindaLoader.show('4'); // Hide the loader VrindaLoader.hide(); // Reinitialize with custom options VrindaLoader.init({ type: '3', // '1', '2', '3', '4' size: 'lg', // 'sm', '', 'lg', 'xl' theme: 'gold', // '', 'gold', 'purple', 'teal', 'red', 'green' autoHide: true, // Auto-hide on page load minDisplayTime: 500 // Minimum display time in ms });
